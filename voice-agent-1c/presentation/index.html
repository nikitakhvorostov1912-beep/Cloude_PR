# services/stt.py — Yandex STT Config

class YandexSTTService:
    BASE_URL = "https://stt.api.cloud.yandex.net"
    SILENCE_RMS_THRESHOLD = 300

    async def recognize_short(self, audio: bytes) -> STTResult:
        """REST API для коротких фраз (< 30 сек)."""
        response = await self._client.post(
            f"{self.BASE_URL}/speech/v1/stt:recognize",
            params={
                "folderId": self._config.folder_id,
                "lang": "ru-RU",
                "sampleRateHertz": 8000,
                "format": "lpcm",
            },
            headers={"Authorization":
                     f"Api-Key {self._config.api_key}"},
            content=audio,
        )
        data = response.json()
        return STTResult(
            text=data.get("result", ""),
            confidence=0.9,
            is_final=True,
        )
